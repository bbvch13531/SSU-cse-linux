classStack{

inttop;
int[]stack;

publicstaticfinalintSTACK_SIZE=10;

publicStack(){
top=-1;
stack=newint[STACK_SIZE];
}

publicintpeek(){
returnstack[top];
}

publicvoidpush(intvalue){
stack[++top]=value;
System.out.printf(stack[top]+"PUSH!\n");
}

publicintpop(){
System.out.printf(stack[top]+"POP!\n");
returnstack[top--];
}

publicvoidprintStack(){
System.out.printf("\n-----STACKLIST-----\n");

for(inti=top;i>=0;i--){
System.out.printf("%d\n",stack[i]);
}

System.out.printf("-----ENDOFLIST-----\n");
}
}
